globals [
  temperature              ;; current global average temperature in °C
  greenhouse-gas-emission  ;; greenhouse gas emission level (ppm CO2)
]

to setup
  clear-all
  reset-ticks
  set temperature 14                ;; baseline modern global temp in °C
  set greenhouse-gas-emission 400  ;; initial emission level (ppm CO2)
  setup-plot
  setup-plot2
end

to go
  increase-emissions
  update-temperature
  update-plot
  update-plot2
  tick
end

to increase-emissions
  ;; Increase emissions by 2 units per tick (year)
  set greenhouse-gas-emission greenhouse-gas-emission + 2
end

to update-temperature
  let sensitivity 0.01 ;; temperature increase per emission unit
  set temperature temperature + (sensitivity * 2)
end

to setup-plot
  set-current-plot "Temperature Over Time"
  set-plot-x-range 0 100
  set-plot-y-range 10 50
  clear-plot
  set-plot-pen-color red
  plot-pen-down
end

to update-plot
  set-current-plot "Temperature Over Time"
  plotxy ticks temperature
end

to setup-plot2
  set-current-plot "Greenhouse Gas Emissions Over Time"
  set-plot-x-range 0 100
  set-plot-y-range 400 1000
  clear-plot
  set-plot-pen-color blue
  plot-pen-down
end

to update-plot2
  set-current-plot "Greenhouse Gas Emissions Over Time"
  plotxy ticks greenhouse-gas-emission
end
